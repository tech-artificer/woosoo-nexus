resources/js/components/AppSidebarHeader.vue
// <script setup> — add imports + state
+ import { Sun, Moon } from 'lucide-vue-next'
+ import { ref, onMounted } from 'vue'

+ const isDark = ref(false)
+ onMounted(() => {
+   isDark.value = document.documentElement.classList.contains('dark')
+ })
+ function toggleDark() {
+   isDark.value = !isDark.value
+   document.documentElement.classList.toggle('dark', isDark.value)
+   localStorage.setItem('appearance', isDark.value ? 'dark' : 'light')
+ }

// Topbar right section — wrap toggle + status
  <div class="hidden items-center gap-1.5 md:flex">
+   <button type="button" @click="toggleDark"
+           class="size-8 rounded-md hover:bg-muted
+                  inline-flex items-center justify-center">
+     <Sun v-if="isDark" class="size-4" />
+     <Moon v-else class="size-4" />
+   </button>
      <div class="flex items-center gap-2 text-xs font-medium text-muted-foreground">
          <span class="inline-flex h-2 w-2 rounded-full bg-emerald-500"></span>
          Secure workspace
      </div>
  </div>

Why radius

rounded-[26px] matches the brand --radius scale used on cards in Step 3, so the page card and its contents share one corner language instead of nesting two different radii.

Why blur

backdrop-blur-sm over xl drops a full-viewport GPU compositing pass on every page — meaningful on the tablet POS hardware — with no visible loss at this opacity.
